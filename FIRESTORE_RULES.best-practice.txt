/**
 * Firestore Security Rules - Production Ready
 * 
 * Copy-paste toàn bộ content này vào:
 * Firebase Console → Firestore Database → Rules tab → Replace all
 * 
 * Sau đó click Publish
 */

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Helper functions
    function isAuthenticated() {
      return request.auth != null;
    }
    
    function isAdmin() {
      return isAuthenticated() && request.auth.token.role == 'admin';
    }
    
    function isSeller() {
      return isAuthenticated() && (request.auth.token.role == 'seller' || request.auth.token.role == 'admin');
    }
    
    function isCurrentUser(userId) {
      return isAuthenticated() && request.auth.uid == userId;
    }

    // ==================== USERS COLLECTION ====================
    match /users/{userId} {
      // Allow owner to read/write own profile
      allow read, write: if isCurrentUser(userId);
      
      // Allow admins to read all users
      allow read: if isAdmin();
      
      // Allow unauthenticated users to read user profiles (non-sensitive data)
      // This is needed for some partner/seller pages to display info
      allow read: if true;
      
      // Allow admins to modify any user
      allow write: if isAdmin();
    }

    // ==================== PRODUCTS COLLECTION ====================
    match /products/{productId} {
      // Allow everyone to read products (public marketplace)
      allow read: if true;
      
      // Allow sellers to create products
      allow create: if isSeller();
      
      // Allow seller (owner) or admin to update/delete their products
      allow update, delete: if (
        isCurrentUser(resource.data.seller_id) ||
        isAdmin()
      );
    }

    // ==================== INVENTORY COLLECTION ====================
    match /inventory/{inventoryId} {
      // Allow everyone to read inventory (stock levels are public)
      allow read: if true;
      
      // Allow sellers to create inventory
      allow create: if isSeller();
      
      // Allow seller (owner) or admin to update/delete
      allow update, delete: if (
        isCurrentUser(resource.data.seller_id) ||
        isAdmin()
      );
    }

    // ==================== ORDERS COLLECTION ====================
    match /orders/{orderId} {
      // Create orders - authenticated users only
      allow create: if isAuthenticated();
      
      // Read own orders or all orders if admin
      allow read: if (
        isCurrentUser(resource.data.user_id) ||
        isAdmin()
      );
      
      // Update/delete own orders or all if admin
      allow update, delete: if (
        isCurrentUser(resource.data.user_id) ||
        isAdmin()
      );
    }

    // ==================== TRANSACTIONS COLLECTION ====================
    match /transactions/{transactionId} {
      // Create transactions - authenticated users only
      allow create: if isAuthenticated();
      
      // Read own transactions or all if admin
      allow read: if (
        isCurrentUser(resource.data.user_id) ||
        isAdmin()
      );
      
      // Only admins can update transactions
      allow update, delete: if isAdmin();
    }

    // ==================== RATINGS COLLECTION ====================
    match /ratings/{ratingId} {
      // Everyone can read ratings
      allow read: if true;
      
      // Authenticated users can create ratings
      allow create: if isAuthenticated();
      
      // Can only update/delete own ratings
      allow update, delete: if isCurrentUser(resource.data.user_id);
    }

    // ==================== DELIVERY TRACKING COLLECTION ====================
    match /deliveries/{deliveryId} {
      // Admin can do everything
      allow read, write: if isAdmin();
      
      // Order owner can read their delivery
      allow read: if isCurrentUser(resource.data.user_id);
      
      // Seller can read deliveries for their products
      allow read: if isSeller();
    }

    // ==================== NOTIFICATIONS COLLECTION ====================
    match /notifications/{notificationId} {
      // User can read their own notifications
      allow read: if isCurrentUser(resource.data.user_id);
      
      // System/admin creates notifications
      allow create: if isAdmin();
      
      // User can delete their own notifications
      allow delete: if isCurrentUser(resource.data.user_id);
    }

    // ==================== SYSTEM SETTINGS ====================
    match /settings/{settingId} {
      // Everyone can read public settings
      allow read: if true;
      
      // Only admins can write settings
      allow write: if isAdmin();
    }

    // ==================== CATCH-ALL ====================
    // Default deny all other collections
    match /{document=**} {
      allow read, write: if isAdmin();
    }
  }
}
